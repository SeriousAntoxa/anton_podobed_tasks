Solution:
let strNamb = (str) => {
  let result = [];
  let arr = str.toLowerCase().split("");
  let newArr = new Set(arr);
  let checkArr = [];
  for (key of newArr) {
    checkArr.push(key);
  }
  arr.forEach((item) => {
    let i = checkArr.findIndex((item2) => item2 === item);
    result.push(i);
  });
  return result
}
